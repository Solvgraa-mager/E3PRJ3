@startuml

actor Bruger #red
participant Motor
participant SumoBot
participant AttackSensor
actor CentralComputer

== Initialization == 

Bruger -> SumoBot: Turn on
SumoBot -> SumoBot: Boot
loop while: !2 SumoBot connCCted
    SumoBot -> CentralComputer: Connects to AP
end

== Game == 
loop while(1)
    CentralComputer -> SumoBot: Speed & direction
    SumoBot -> SumoBot: Processing 
    SumoBot -> Motor: Set speed & direction
group interrupt
AttackSensor -> SumoBot: Interrupt, attack
SumoBot -> CentralComputer: Attack!
    CentralComputer -> CentralComputer: Substract life
end
end

@enduml
